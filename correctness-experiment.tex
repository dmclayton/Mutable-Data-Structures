    \oraclev{$\REPO(\col)$}\\[2pt]
      $\pub \getsr \Rep_\key(\col)$\\
      if $\pub = \bot$ return $\bot$\\

      $\pub_\ct \gets \pub$\\
      $\setC_\ct \gets \emptyset$\\

    \oraclev{$\REVO(i)$}\\[2pt]
     $\setP \gets \setP \cup \{i\}$ \\
      return $\pub_i$
  }

      $\col_i \gets \up(X)$\\
      $\pub_i \getsr \pub$\\
      $\setC_i \gets \emptyset$\\
